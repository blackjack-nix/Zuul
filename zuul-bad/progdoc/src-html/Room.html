008import java.util.HashMap; 
009import java.util.Set;
010import java.util.Iterator;
011
012public class Room
013{
014    public HashMap <String , Room>  aSortieHM;
015    private String aDescription = "Cette salle est la salle la plus au sud du temple. Elle est gardée par 2 gardes ainsi qu'un garde d'élite.";//description succinte de la salle
016
017    /**
018     * Create rooms with a string description
019     *  and create a HashMap of the exits for each
020     */
021    public Room (final String pDescription){
022        this.aDescription = pDescription ;
023        aSortieHM = new HashMap <String , Room> ();
024    }//Room(.)
025
026    /**
027     * return a string of the room description
028     * @return  Return a String description of the room
029     */
030    public String getDescription (){
031        return this.aDescription;
032    }//getDescription(.)
033
034    /**
035     * change the exit of a room
036     * @param pDirection A String for the direection you want to go
037     * @param pRoom Change the room
038     */
039    public void setExits (final String pDirection, final Room pRoom){
040        aSortieHM.put(pDirection,pRoom);
041    }//setExit(....)
042
043    /**
044     * Return the exit of the room
045     * @return Return the room after taking the direction
046     * @param pDirection Direction you want to take in the room
047     */
048    public Room getExit(final String pDirection){
049        return aSortieHM.get(pDirection);
050    }//getExit(.)
051
052    /**
053     * @return Return a string description of the room
054     */
055    public String getExitsString(){
056        String vSortie = "";
057        Set<String> keys = aSortieHM.keySet();
058        for ( String aSortieHM : keys) vSortie += ' ' + aSortieHM;
059        return vSortie;       
060    } //getExitsString()
061    
062    /**
063     * Renvoie une description détillée de cette piece sous la forme :
064     *      vous etes dans + piece
065     *      sorties : 
066     * @return Une description de la piece avec les sorties
067     */
068    public String getLongDescription(){
069        return "Vous etes "+this.aDescription+". Sorties : "+this.getExitsString();
070    }
071} // Room
