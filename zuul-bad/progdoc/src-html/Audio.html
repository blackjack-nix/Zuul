001import javax.sound.sampled.*;
002/**
003 * Class for the Sound of the project Zuul-Wars
004 * @author Théo Péresse et Quentin Martins
005 * @version finale
006 * Available on GitHub
007 */ 
008public class Audio {
009    // ## Attributs ##
010    private Clip aClip;
011    
012    // ## Constructor ##
013    
014    /**
015     * Contructeur de l'audio
016     * @param pSon le chemin vers l'audio. ! ne prend que les .wav
017     */
018    public Audio (final String pSon){
019        try {
020            AudioInputStream vAudio = AudioSystem.getAudioInputStream(getClass().getResource(pSon));
021            this.aClip = AudioSystem.getClip();
022            this.aClip.open(vAudio);                       
023        }
024        catch (final Exception pEx){
025            System.out.println("Erreur son");            
026        }              
027    }          
028    
029    /**
030     * lance le son sur les enceintes du pc
031     */
032    public void play() {this.aClip.start();}
033    
034    /**
035     * Coupe le son en cours
036     */
037    public void stop() {this.aClip.stop();}
038    
039    /**
040     * creer un son puis le lance
041     * @param pSon String qui correspond au chemin du son
042     */
043    public static void playSound (final String pSon){
044        Audio vS = new Audio(pSon);
045        vS.play();
046    }
047    /**
048     * Renvoie le boolean qui vaut vrai si la musique est finie et faux si elle est en cours
049     * @return Boolean estFinit
050     */
051    public boolean isFinished(){
052        return ! aClip.isActive();        
053    }
054}
