001/**
002 * Class for the Doors of the project Zuul-Wars
003 * @author Théo Péresse
004 * @version finale
005 * Available on GitHub
006 */ 
007 
008public class Door
009{
010    // ## Attributs ##
011    private boolean isLocked;
012    private boolean isTrap;
013    private boolean isGoodGirection;
014    
015    
016    // ## Constructor ## 
017    
018    /**
019     * constructeur naturel
020     * @param pLocked boolean qui dit si la door est locked
021     * @param pTrap boolean qui dit si la door est trap
022     * @param pGoodDirection (si trop only) boolean qui dit si bonne direction de passage
023     */
024    public Door (final boolean pLocked , final boolean pTrap , final boolean pGoodDirection){
025        this.isLocked = pLocked;
026        this.isTrap = pTrap;
027        this.isGoodGirection = pGoodDirection;        
028    }
029    
030    
031    // ## Accesseur ##
032    
033    /**
034     * Accesseur de lock
035     * @return boolean si la door est locked
036     */
037    public boolean isLocked(){
038        return this.isLocked;
039    }
040    
041    /**
042     * Accesseur de trap
043     * @return boolean si la door est trap
044     */
045    public boolean isTrap(){
046        return this.isTrap;
047    }
048    
049    /**
050     * Accesseur de cango
051     * @return boolean si la door trap peut etre franchie dans ce sens
052     */
053    public boolean canGo(){
054        return this.isGoodGirection;
055    }
056    
057    
058    // ## Modificateurs ##
059    
060    /**
061     * setteur de locked. Permet de deverouiller une piece
062     * @param pLocked permet de debloquer une door 
063     */
064    public void setLocked (final boolean pLocked){
065        this.isLocked = pLocked;
066    }
067    
068    /**
069     * setteur de cango
070     * @param pGoodDirection boolean qui permet de set gooddirection
071     */
072    public void setCanGo(final boolean pGoodDirection){
073        this.isGoodGirection = pGoodDirection;
074    }
075}
